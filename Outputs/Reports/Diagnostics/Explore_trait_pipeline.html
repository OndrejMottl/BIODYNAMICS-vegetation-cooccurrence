Section 2 — Segment 2: Raw QC flags

Show code 
# Auto-detect most recent QC report in Data/Temp/.
vec_qc_report_paths <-
  fs::dir_ls(
    here::here("Data/Temp"),
    regexp = "trait_qc_report_\\d{4}-\\d{2}-\\d{2}\\.csv$"
  )

if (
  base::length(vec_qc_report_paths) == 0L
) {
  cli::cli_warn(
    c(
      "No trait_qc_report_*.csv found in Data/Temp/.",
      "i" = "Run the traits pipeline (Segment 2) to generate the report.",
      "i" = "Section 2 outputs will be skipped."
    )
  )

  data_qc_report <- NULL
} else {
  path_qc_report <-
    vec_qc_report_paths |>
    base::sort() |>
    utils::tail(1L)

  base::message(
    stringr::str_glue("Using QC report: {base::basename(path_qc_report)}")
  )

  data_qc_report <-
    readr::read_csv(
      path_qc_report,
      show_col_types = FALSE
    ) |>
    dplyr::mutate(
      outlier_fraction =
        .data[["n_suspected_outliers_taxon"]] / .data[["n_records"]]
    )
}

# Branched targets (tar_map) cannot be read as a combined object via
# tar_read() when the branches return non-stackable types (dist, hclust).
# Read each branch individually using branch names from the pipeline metadata,
# then collect into a named list keyed by continent.
data_meta_pipeline <-
  targets::tar_meta(store = path_traits_store)

# Read all branches of a pattern target as a named list (one element per
# continent, in the same order as vec_continent_ids).
continent_branches <- function(pattern_name) {
  vec_children <-
    data_meta_pipeline |>
    dplyr::filter(.data[["name"]] == pattern_name) |>
    dplyr::pull("children") |>
    purrr::chuck(1L)

  vec_children |>
    rlang::set_names(vec_continent_ids) |>
    purrr::map(
      .f = ~ targets::tar_read_raw(.x, store = path_traits_store)
    )
}
