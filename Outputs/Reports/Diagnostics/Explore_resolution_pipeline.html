3.1 R² comparison across resolutions

Show code 
if (
  base::length(list_model_evaluation) > 0L
) {
  list_model_evaluation |>
    purrr::imap(
      .f = ~ {
        tibble::tibble(
          resolution = .y,
          r2_mcfadden = purrr::pluck(.x, "model", "R2-McFadden"),
          r2_nagelkerke = purrr::pluck(.x, "model", "R2-Nagelkerke")
        )
      }
    ) |>
    purrr::list_rbind() |>
    tidyr::pivot_longer(
      cols = c("r2_mcfadden", "r2_nagelkerke"),
      names_to = "metric",
      values_to = "value"
    ) |>
    dplyr::mutate(
      metric = dplyr::case_match(
        .data[["metric"]],
        "r2_mcfadden" ~ "McFadden R²",
        "r2_nagelkerke" ~ "Nagelkerke R²"
      )
    ) |>
    ggplot2::ggplot(
      ggplot2::aes(
        x = .data[["resolution"]],
        y = .data[["value"]],
        fill = .data[["resolution"]]
      )
    ) +
    ggplot2::geom_col(alpha = 0.7) +
    ggplot2::facet_wrap(ggplot2::vars(.data[["metric"]])) +
    ggplot2::labs(
      x = "Taxonomic resolution",
      y = "R²",
      fill = NULL
    ) +
    ggplot2::theme_minimal() +
    ggplot2::theme(legend.position = "none")
} else {
  base::message("No model_evaluation targets found — Section 3 skipped.")
